<div><font face="&#39;courier new&#39;, monospace">    /usr/lib/rabbitmq/bin/${SCRIPT} &quot;$@&quot;</font></div><div><font face="&#39;courier new&#39;, monospace">elif [ `id -u` = 0 ] ; then</font></div><div><font face="&#39;courier new&#39;, monospace">    # don&#39;t use su, use start-stop-daemon instead</font></div>
<div><font face="&#39;courier new&#39;, monospace">    #su rabbitmq -s /bin/sh -c &quot;/usr/lib/rabbitmq/bin/${SCRIPT} ${CMDLINE}&quot;</font></div><div><font face="&#39;courier new&#39;, monospace">    start-stop-daemon \</font></div>
