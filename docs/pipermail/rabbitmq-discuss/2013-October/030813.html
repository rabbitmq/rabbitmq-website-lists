    % unknown_ca errors are silently ignored prior to R14B unless we

        verify_none -> SslOptsConfig;
        verify_peer -> [{verify_fun, fun([])    -> true;
                                        ([_|_]) -> false

                        | SslOptsConfig]
    end.
